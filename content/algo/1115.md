## 解法一： Semaphore

```java
class FooBar {
    private int n;

    Semaphore foo = new Semaphore(1);

    Semaphore bar = new Semaphore(0);

    public FooBar(int n) {
        this.n = n;
    }

    public void foo(Runnable printFoo) throws InterruptedException {

        for (int i = 0; i < n; i++) {

            // printFoo.run() outputs "foo". Do not change or remove this line.
            foo.acquire();
            printFoo.run();
            bar.release();
        }
    }

    public void bar(Runnable printBar) throws InterruptedException {

        for (int i = 0; i < n; i++) {

            // printBar.run() outputs "bar". Do not change or remove this line.
            bar.acquire();;
            printBar.run();
            foo.release();
        }
    }
}
```

## 解法二：Spin Lock

```java
class FooBar {
    private int n;

    private volatile boolean flag = false;

    public FooBar(int n) {
        this.n = n;
    }

    public void foo(Runnable printFoo) throws InterruptedException {

        for (int i = 0; i < n; i++) {

            // printFoo.run() outputs "foo". Do not change or remove this line.
            // 如果while的循环体里面不加yield会导致超时，它可能不会自己让出cpu
            while (flag) {Thread.yield();}
            printFoo.run();
            flag = !flag;
        }
    }

    public void bar(Runnable printBar) throws InterruptedException {

        for (int i = 0; i < n; i++) {
            // printBar.run() outputs "bar". Do not change or remove this line.
            while (!flag) {Thread.yield();}
            printBar.run();
            flag = !flag;
        }
    }
}
```

## 解法三：Lock + Conditon

```java
class FooBar {
    private int n;

    ReentrantLock lock = new ReentrantLock();

    Condition foo = lock.newCondition();

    Condition bar = lock.newCondition();

    volatile boolean flag = false;

    public FooBar(int n) {
        this.n = n;
    }

    public void foo(Runnable printFoo) throws InterruptedException {

        for (int i = 0; i < n; i++) {

            // printFoo.run() outputs "foo". Do not change or remove this line.
            lock.lock();
            if (flag) {
                foo.await();
            }
            printFoo.run();
            flag = !flag;
            bar.signal();
            lock.unlock();
        }
    }

    public void bar(Runnable printBar) throws InterruptedException {

        for (int i = 0; i < n; i++) {

            // printBar.run() outputs "bar". Do not change or remove this line.
            lock.lock();
            if (!flag) {
                bar.await();
            }
            printBar.run();
            flag = !flag;
            foo.signal();
            lock.unlock();
        }
    }
}
```

## 解法四：LockSupport

```java
class FooBar {
    private int n;

    private boolean flag = true;

    Thread foo = null;

    Thread bar = null;

    public FooBar(int n) {
        this.n = n;
    }

    public void foo(Runnable printFoo) throws InterruptedException {

        foo = Thread.currentThread();
        for (int i = 0; i < n; i++) {
            if (!flag) {
                LockSupport.park();
            }
            // printFoo.run() outputs "foo". Do not change or remove this line.
            printFoo.run();
            flag = !flag;
            LockSupport.unpark(bar);
        }
    }

    public void bar(Runnable printBar) throws InterruptedException {

        bar = Thread.currentThread();
        for (int i = 0; i < n; i++) {

            // printBar.run() outputs "bar". Do not change or remove this line.
            if (flag) {
                LockSupport.park();
            }
            printBar.run();
            flag = !flag;
            LockSupport.unpark(foo);
            ;
        }
    }
}
```

## 解法五：Synchronized + wait/notifyAll

```java
class FooBar {
    private int n;

    Object lock = new Object();

    volatile boolean flag = false;

    public FooBar(int n) {
        this.n = n;
    }

    public void foo(Runnable printFoo) throws InterruptedException {

        for (int i = 0; i < n; i++) {
            // printFoo.run() outputs "foo". Do not change or remove this line.
            synchronized (lock) {
                if (flag) {
                    lock.wait();
                }
                printFoo.run();
                flag = !flag;
                lock.notifyAll();
            }
        }
    }

    public void bar(Runnable printBar) throws InterruptedException {

        for (int i = 0; i < n; i++) {

            // printBar.run() outputs "bar". Do not change or remove this line.
            synchronized (lock) {
                if (!flag){
                    lock.wait();
                }
                printBar.run();
                flag=!flag;
                lock.notifyAll();
            }
        }
    }
}
```

## 解法六：CyclicBarrier

```java
class FooBar {
    private int n;

    CyclicBarrier barrier = new CyclicBarrier(2);

    volatile boolean flag = false;

    public FooBar(int n) {
        this.n = n;
    }

    public void foo(Runnable printFoo) throws InterruptedException {

        for (int i = 0; i < n; i++) {
            // printFoo.run() outputs "foo". Do not change or remove this line.
            while (flag) {
                Thread.yield();
            }
            printFoo.run();
            flag = !flag;
            try {
                barrier.await();
            } catch (BrokenBarrierException e) {
                e.printStackTrace();
            }
        }
    }

    public void bar(Runnable printBar) throws InterruptedException {

        for (int i = 0; i < n; i++) {

            // printBar.run() outputs "bar". Do not change or remove this line.
            try {
                barrier.await();
            } catch (BrokenBarrierException e) {
                e.printStackTrace();
            }
            printBar.run();
            flag = !flag;
        }
    }
}
```
