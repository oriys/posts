## 解法一：Spin Lock

```java
class ZeroEvenOdd {
    private int n;

    volatile int count = 0;

    public ZeroEvenOdd(int n) {
        this.n = n;
    }

    // printNumber.accept(x) outputs "x", where x is an integer.
    public void zero(IntConsumer printNumber) throws InterruptedException {
        for (int i = 0; i < n; i++) {
            while (count % 4 == 1 || count % 4 == 3) {
                Thread.yield();
            }
            printNumber.accept(0);
            count++;
        }
    }

    public void odd(IntConsumer printNumber) throws InterruptedException {
        for (int i = 1; i <= n; i += 2) {
            while (count % 4 != 1) {
                Thread.yield();
            }
            printNumber.accept(i);
            count++;
        }
    }

    public void even(IntConsumer printNumber) throws InterruptedException {
        for (int i = 2; i <= n; i += 2) {
            while (count % 4 != 3) {
                Thread.yield();
            }
            printNumber.accept(i);
            count++;
        }
    }
}
```

## 解法二：Semaphore

```java
class ZeroEvenOdd {
    private int n;

    Semaphore zero = new Semaphore(1);

    Semaphore even = new Semaphore(0);

    Semaphore odd = new Semaphore(0);

    public ZeroEvenOdd(int n) {
        this.n = n;
    }

    // printNumber.accept(x) outputs "x", where x is an integer.
    public void zero(IntConsumer printNumber) throws InterruptedException {
        for (int i = 0; i < n; i++) {
            zero.acquire();
            printNumber.accept(0);
            if (i % 2 == 0) {
                odd.release();
            }
            else {
                even.release();
            }
        }
    }

    public void even(IntConsumer printNumber) throws InterruptedException {
        for (int i = 2; i <= n; i += 2) {
            even.acquire();
            printNumber.accept(i);
            zero.release();
        }
    }

    public void odd(IntConsumer printNumber) throws InterruptedException {
        for (int i = 1; i <= n; i += 2) {
            odd.acquire();
            printNumber.accept(i);
            zero.release();
        }
    }
}
```

## 解法三：CyclicBarrier

```java
class ZeroEvenOdd {
    private int n;

    CyclicBarrier zero = new CyclicBarrier(2);

    CyclicBarrier even = new CyclicBarrier(2);

    CyclicBarrier odd = new CyclicBarrier(2);

    public ZeroEvenOdd(int n) {
        this.n = n;
    }

    // printNumber.accept(x) outputs "x", where x is an integer.
    public void zero(IntConsumer printNumber) throws InterruptedException {
        for (int i = 0; i < n; i++) {
            try {
                printNumber.accept(0);
                if (i % 2 == 0) {
                    odd.await();
                }
                else {
                    even.await();
                }
                zero.await();
            } catch (BrokenBarrierException e) {
                e.printStackTrace();
            }
        }
    }

    public void even(IntConsumer printNumber) throws InterruptedException {
        for (int i = 2; i <= n; i += 2) {
            try {
                even.await();
                printNumber.accept(i);
                zero.await();
            } catch (BrokenBarrierException e) {
                e.printStackTrace();
            }
        }
    }

    public void odd(IntConsumer printNumber) throws InterruptedException {
        for (int i = 1; i <= n; i += 2) {
            try {
                odd.await();
                printNumber.accept(i);
                zero.await();
            } catch (BrokenBarrierException e) {
                e.printStackTrace();
            }
        }
    }
}
```

## 解法四：LockSupport

```java
class ZeroEvenOdd {
    private int n;

    Thread zero = null;

    Thread odd = null;

    Thread even = null;

    public ZeroEvenOdd(int n) {
        this.n = n;
    }

    // printNumber.accept(x) outputs "x", where x is an integer.
    public void zero(IntConsumer printNumber) throws InterruptedException {
        zero = Thread.currentThread();
        for (int i = 0; i < n; i++) {
            printNumber.accept(0);
            // 不判空在leetcode上会出现超时，一直要到odd和even初始化完再往下走
            while (odd == null || even == null) {
                Thread.yield();
            }
            if (i % 2 == 0) {
                LockSupport.unpark(odd);
            }
            else {
                LockSupport.unpark(even);
            }
            LockSupport.park();
        }
    }

    public void even(IntConsumer printNumber) throws InterruptedException {
        even = Thread.currentThread();
        for (int i = 2; i <= n; i += 2) {
            LockSupport.park();
            printNumber.accept(i);
            LockSupport.unpark(zero);
        }
    }

    public void odd(IntConsumer printNumber) throws InterruptedException {
        odd = Thread.currentThread();
        for (int i = 1; i <= n; i += 2) {
            LockSupport.park();
            printNumber.accept(i);
            LockSupport.unpark(zero);
        }
    }
}
```

## 解法五：Synchroinzed + wait/notifyAll

```java
class ZeroEvenOdd {
    private int n;

    volatile int count = 0;

    final Object lock = new Object();

    public ZeroEvenOdd(int n) {
        this.n = n;
    }

    // printNumber.accept(x) outputs "x", where x is an integer.
    public void zero(IntConsumer printNumber) throws InterruptedException {
        for (int i = 0; i < n; i++) {
            synchronized (lock) {
                while (count != 0) {
                    lock.wait();
                }
                printNumber.accept(0);
                count = i % 2 + 1;
                lock.notifyAll();
            }
        }
    }

    public void even(IntConsumer printNumber) throws InterruptedException {
        for (int i = 2; i <= n; i += 2) {
            synchronized (lock) {
                while (count != 2) {
                    lock.wait();
                }
                printNumber.accept(i);
                count = 0;
                lock.notifyAll();
            }
        }
    }

    public void odd(IntConsumer printNumber) throws InterruptedException {
        for (int i = 1; i <= n; i += 2) {
            synchronized (lock) {
                while (count != 1) {
                    lock.wait();
                }
                printNumber.accept(i);
                count = 0;
                lock.notifyAll();
            }
        }
    }
}
```

### 解法六：Lock + Condition

```java
class ZeroEvenOdd {
    private int n;

    ReentrantLock lock = new ReentrantLock();

    Condition zero = lock.newCondition();

    Condition odd = lock.newCondition();

    Condition even = lock.newCondition();

    volatile int count = 0;

    public ZeroEvenOdd(int n) {
        this.n = n;
    }

    // printNumber.accept(x) outputs "x", where x is an integer.
    public void zero(IntConsumer printNumber) throws InterruptedException {
        for (int i = 0; i < n; i++) {
            lock.lock();
            while (count != 0) {
                zero.await();
            }
            printNumber.accept(0);
            if (i % 2 == 0) {
                count = 1;
                odd.signal();
            }
            else {
                count = 2;
                even.signal();
            }
            lock.unlock();
        }
    }

    public void even(IntConsumer printNumber) throws InterruptedException {
        for (int i = 2; i <= n; i += 2) {
            lock.lock();
            if (count != 2) {
                even.await();
            }
            printNumber.accept(i);
            count = 0;
            zero.signal();
            lock.unlock();
        }
    }

    public void odd(IntConsumer printNumber) throws InterruptedException {
        for (int i = 1; i <= n; i += 2) {
            lock.lock();
            if (count != 1) {
                odd.await();
            }
            printNumber.accept(i);
            count = 0;
            zero.signal();
            lock.unlock();
        }
    }
}
```
